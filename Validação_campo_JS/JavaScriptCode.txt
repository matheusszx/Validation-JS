        <label for="id">Id. Único</label>&nbsp;
        <input type="text" id="identificador" onchange="myFunction()" required />
    </div>

    <script>
        function myFunction() {

            //Chamando o ID do input como variavel
            var idUserName = document.getElementById("identificador");

            //Remove os espaços em brancos
            idUserName.value = idUserName.value.replace(/\s/g, "");

            //caracteres válidos do campo #Identificador
            idUserName.value = idUserName.value.replace(/[^a-zA-Z0-9]/g, "");
            return true;
        }
    </script>
